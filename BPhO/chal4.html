import matplotlib.pyplot as plt import numpy as np import math fig, ax = plt.subplots() u = 10 h = 2 g = 9.81 theta = 60 x = np.linspace(0, 13, 100) sinTheta = math.sin(math.radians(theta)) cosTheta = math.cos(math.radians(theta)) tanTheta = math.tan(math.radians(theta)) yNormal = h + x * tanTheta - (g*np.square(x)*(1+tanTheta**2))/(2*u**2) R_x = ((u**2)/g) * (sinTheta * cosTheta + cosTheta * math.sqrt(sinTheta**2 + (2*g*h)/(u**2))) R_y = np.interp(R_x, x, yNormal) plt.plot(R_x, R_y, marker="x", markersize=10, markeredgecolor="red", markeredgewidth=2, zorder=2) thetaMax = math.asin(1/(math.sqrt(2+2*g*h/(u**2)))) RMax = ((u**2)/g) * math.sqrt(1+(2*g*h)/u**2) tanThetaMax = math.tan(thetaMax) yMax = h + x * tanThetaMax - (g*np.square(x)*(1+tanThetaMax**2))/(2*u**2) plt.plot(x, yMax, color='red', linestyle='dashed') plt.plot(x, yNormal, color='blue', zorder=1) plt.title('') plt.xlabel('x /m') plt.ylabel('y /m') plt.xlim(0, 14) plt.ylim(0, 6) plt.grid() fig
