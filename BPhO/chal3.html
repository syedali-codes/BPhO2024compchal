import matplotlib.pyplot as plt import numpy as np import math fig, ax = plt.subplots() X = 1000 Y = 300 u = 150 h = 0 g = 9.81 x = np.linspace(0, X, 100) a = (g*X**2)/(2*u**2) b = -X c = Y - h + (g*X**2)/(2*u**2) thetaHigh = math.atan((-b+math.sqrt((b**2)-4*a*c))/(2*a)) thetaLow = math.atan((-b-math.sqrt((b**2)-4*a*c))/(2*a)) thetaMin = math.atan((Y + math.sqrt((Y**2)+(X**2)))/X) uMin = math.sqrt(g) * math.sqrt(Y+math.sqrt(X**2+Y**2)) yMin = h + x * math.tan(thetaMin) - (g*np.square(x)*(1+math.tan(thetaMin)**2))/(2*uMin**2) yHigh = h + x * math.tan(thetaHigh) - (g*np.square(x)*(1+math.tan(thetaHigh)**2))/(2*u**2) yLow = h + x * math.tan(thetaLow) - (g*np.square(x)*(1+math.tan(thetaLow)**2))/(2*u**2) plt.plot(x, yHigh, color='blue', linewidth=2.5) plt.plot(x, yLow, color='orange', linewidth=2.5) plt.plot(x, yMin, color='gray', linewidth=2.5) plt.plot(X, Y, marker=".", markersize=10, markeredgecolor="black", markerfacecolor="yellow") plt.plot(1030, 220, marker="$( X,Y )$", markersize=30, markeredgecolor="black", markerfacecolor="black") plt.title('Projectile to hit X,Y') plt.xlabel('x /m') plt.ylabel('y above launch height(0m) /m') plt.xlim(0, 1200) plt.ylim(0,1200.00) plt.grid() fig
